% Diapositiva 1: Introducción
\begin{frame}{¿Qué es NotebookLM?}
    \textbf{NotebookLM} es un asistente de investigación personalizado potenciado por inteligencia artificial (Gemini 1.5 Pro).
    \vspace{0.5cm}
    \begin{itemize}
        \item \textbf{Enfoque en fuentes:} A diferencia de otros chatbots, se basa exclusivamente en los documentos que tú subes.
        \item \textbf{Privacidad:} La IA no utiliza tus datos personales o archivos para entrenar modelos públicos.
        \item \textbf{Comprensión contextual:} Capacidad para analizar múltiples archivos (PDF, texto, Google Docs) simultáneamente.
    \end{itemize}
\end{frame}

% Diapositiva 2: Funciones Principales
\begin{frame}{Funciones Clave y Valor Añadido}
    \begin{block}{Herramientas de Productividad}
        \begin{itemize}
            \item \textbf{Citas directas:} Cada respuesta incluye enlaces a la fuente exacta dentro de tus documentos.
            \item \textbf{Resúmenes automáticos:} Genera guías de estudio, notas estructuradas y esquemas de contenido.
            \item \textbf{Audio Overview:} Convierte tus notas en una conversación de audio tipo ``podcast'' entre dos expertos.
        \end{itemize}
    \end{block}
    \vspace{0.3cm}
    \textit{``De la lectura pasiva a la colaboración activa con la información."}
\end{frame}
